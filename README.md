## Usage

### ES6

```javascript
import Constant from 'go-constant'

const SIZE = {
	SMALL: Constant(3, "Small"),
	MEDIUM: Constant(5, "Medium"),
	LARGE: Constant(8, "Large")
};

 
let item = { size: 6 };

let message;

if (item.size > SIZE.MEDIUM) {
	message = "Item is too big!;
}

console.log(message); // => Item is too big!

```
